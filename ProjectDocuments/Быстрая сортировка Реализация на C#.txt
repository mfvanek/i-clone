class QuickSorting {
   public static void sorting(double[] arr, long first, long last) {
      double p = arr[(last - first)2 + first];
      double temp;
      long i = first, j = last;
      while(i = j) {
         while(arr[i]  p && i = last)  ++i;
         while(arr[j]  p && j = first) --j;
            if(i = j) {
               temp = arr[i];
               arr[i] = arr[j];
               arr[j] = temp;
               ++i; --j;
            }
      }
      if(j  first) sorting(arr, first, j);
      if(i  last)  sorting(arr, i, last);
   }
}
